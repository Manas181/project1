import RPi.GPIO as GPIO
import time

# Set GPIO mode and pin number
GPIO.setmode(GPIO.BOARD)
servo_pin = 18

# Set up the GPIO pin for output
GPIO.setup(servo_pin, GPIO.OUT)

# Create a PWM object with a frequency of 50 Hz
pwm = GPIO.PWM(servo_pin, 50)

# Start the PWM at 0% duty cycle
pwm.start(0)

try:
    while True:
        # Move the servo to 0 degrees
        pwm.ChangeDutyCycle(2.5)
        time.sleep(1)

        # Move the servo to 90 degrees
        pwm.ChangeDutyCycle(7.5)
        time.sleep(1)

        # Move the servo back to 0 degrees
        pwm.ChangeDutyCycle(2.5)
        time.sleep(1)

        # Wait for 7 seconds before starting again
        time.sleep(7)

except KeyboardInterrupt:
    # Clean up the GPIO pins on exit
    pwm.stop()
    GPIO.cleanup()
